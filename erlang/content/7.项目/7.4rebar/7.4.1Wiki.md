## 下载

```bash
wget https://s3.amazonaws.com/rebar3/rebar3
chmod 777 rebar3
./rebar3

./rebar3 local install #全局环境变量
```

## Creating a New Project

```bash
./rebar3 new release myrelease

vim rebar.config
```

```config
{erl_opts, [debug_info]}.

{deps, [
    {cowboy, {git, "https://github.com/ninenines/cowboy", {tag, "2.4.0"}}},
    {jsx, {git, "https://github.com/talentdeficit/jsx", {tag, "v2.9.0"}}}
]}.

{relx, [
    {release, {myrel, "0.1.0"}, [
        myrel,
        cowboy,
        jsx,
        sasl,
        mnesia
    ]},

    {sys_config, "./config/sys.config"},
    {vm_args, "./config/vm.args"},

    {dev_mode, true},
    {include_erts, false},

    {extended_start_script, true}
]}.

{profiles, [
    {prod, [
        {relx, [
            {sys_config_src, "./config/sys.config.src"},
            {vm_args_src, "./config/vm.args.src"},
            {dev_mode, false},
            {include_erts, true},
            {system_libs, true},
            {include_src, false},
            {debug_info, strip}
        ]}
    ]}
]}.
```

```bash
./rebar3 compile
```

## 打包

```bash
./rebar3 new release myrel
./rebar3 release
./rebar3 as prod tar

```
