
1.check_cookie( A ) ->
    H = A#arg.headers,
    case yaws_api:find_cookie_val( ?Cook, H#headers.cookie ) of
        Val when Val /= [] ->
            case yaws_api:cookieval_to_opaque( Val ) of
                {ok, Sess} ->
                    {ok, Sess, Val};
                { error, { has_session, Sess }} ->
                    {ok, Sess};
                Else ->
                    Else
            end;
        [] ->
            {error, nocookie}
    end.
2.{ok, _Sess, Cookie} -> yaws_api:delete_cookie_session( Cookie ); 删除session
