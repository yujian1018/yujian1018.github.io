#include <iostream>

using namespace std;

template<typename M, typename N>
class point {
private:
    M x;
    N y;
public:
    point() : x(0), y(0) {}

    point(M x, N y) : x(x), y(y) {}

    void set_xy(M x, N y);

    M get_x();

    N get_y();
};

template<typename M, typename N>
void point<M, N>::set_xy(M x, N y) {
    this->x = x;
    this->y = y;
}

template<typename M, typename N>
M point<M, N>::get_x() {
    return x;
}
int main() {
    point<int, float > p(10, 12.3);
     Point<float, float> *p = new Point<float, int>(10.6, 109);
    cout << p.get_x() << endl;

}
